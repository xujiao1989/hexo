    • 完成方法count(str)，传入字符串，返回字符串中出现次数最多的字符及次数

      function count(str){

          var length = str.length;
    if(length ==0){
        return;
    }
    var max = 1;
    var maxChar = str[0];
    var obj = {};
    for(var i = 0;i<length;i++){
        if(str[i] in obj){
            obj[str[i]] ++;
            if(max < obj[str[i]]){
                max = obj[str[i]];
                maxChar = str[i];
            }
        }else{
            obj[str[i]] = 1;
        }
    }
   return {
       maxNum : max,
       maxChar :maxChar
   }
}
